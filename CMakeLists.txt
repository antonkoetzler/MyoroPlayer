CMAKE_MINIMUM_REQUIRED(VERSION 2.0)
PROJECT(MyoroPlayer)

FIND_PACKAGE(wxWidgets REQUIRED core base)
INCLUDE("${wxWidgets_USE_FILE}")

AUX_SOURCE_DIRECTORY(./src/ SRC)
INCLUDE_DIRECTORIES(./include/)

ADD_EXECUTABLE(
  ${PROJECT_NAME}
  ${SRC}
)

TARGET_LINK_LIBRARIES(
  ${PROJECT_NAME}
  ${wxWidgets_LIBRARIES}
)
